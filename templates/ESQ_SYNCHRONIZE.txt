-- Genera SYNCHRONIZE TABLA
set serveroutput on
BEGIN
        DBMS_OUTPUT.NEW_LINE;
        DBMS_OUTPUT.PUT_LINE('Inicio: ' || TO_CHAR(SYSDATE,'YYYY/MM/DD  HH24:MI:SS'));

	DBMS_REDEFINITION.SYNC_INTERIM_TABLE(
	                                      uname      => 'ESQUEMA', 
					      orig_table => 'TABLA', 
					      int_table  => 'I_TABLA'
					    );

        DBMS_OUTPUT.PUT_LINE('Proceso SINCHRONYZE TABLA : OK');
        DBMS_OUTPUT.PUT_LINE('Finalización: ' || TO_CHAR(SYSDATE,'YYYY/MM/DD  HH24:MI:SS'));
        DBMS_OUTPUT.NEW_LINE;



EXCEPTION
   WHEN others then
	DBMS_OUTPUT.PUT_LINE('Proceso SINCHRONYZE TABLA  con error: '||SQLERRM);
	DBMS_OUTPUT.PUT_LINE('Finalización: ' || TO_CHAR(SYSDATE,'YYYY/MM/DD  HH24:MI:SS'));
        DBMS_OUTPUT.NEW_LINE;
END;
/
EXIT
