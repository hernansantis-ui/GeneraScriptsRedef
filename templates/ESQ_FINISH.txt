-- Genera FINISH REDEFINITION TABLA
set serveroutput on

BEGIN
        DBMS_OUTPUT.NEW_LINE;
        DBMS_OUTPUT.PUT_LINE('Inicio: ' || TO_CHAR(SYSDATE,'YYYY/MM/DD  HH24:MI:SS'));

        DBMS_REDEFINITION.FINISH_REDEF_TABLE(
					      uname      => 'ESQUEMA' , 
					      orig_table => 'TABLA' , 
					      int_table  => 'I_TABLA'
					    );

        DBMS_OUTPUT.PUT_LINE('Proceso FINISH TABLA : OK');
        DBMS_OUTPUT.PUT_LINE('Finalización: ' || TO_CHAR(SYSDATE,'YYYY/MM/DD  HH24:MI:SS'));
        DBMS_OUTPUT.NEW_LINE;
        
        FOR INDICE IN (SELECT INDEX_NAME FROM ALL_INDEXES 
                       WHERE TABLE_NAME='TABLA' AND OWNER='ESQUEMA')
        LOOP
                EXECUTE IMMEDIATE('ALTER INDEX ESQUEMA.INDICE NOPARALLEL') ;
        END LOOP;        

EXCEPTION
    when OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Proceso FINISH TABLA con error: '||SQLERRM);
        DBMS_OUTPUT.PUT_LINE('Finalización: ' || TO_CHAR(SYSDATE,'YYYY/MM/DD  HH24:MI:SS'));
        DBMS_OUTPUT.NEW_LINE;

END;
/
EXIT
